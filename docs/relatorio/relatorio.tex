  \ifcsdef{labelused@#1}{%
    \PackageError{requisitos}{Label duplicado: #1}{Cada requisito deve ter um label unico.}%
  }{%
    \global\csdef{labelused@#1}{}%\item \textbf{Visão de Dados do Software}
  }%
}
